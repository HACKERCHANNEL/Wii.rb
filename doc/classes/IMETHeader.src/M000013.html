        def add(sizes, names)
                @tag = "IMET"
                @size = 0x00000600
                @unk = 00000003
                @sizes[0..2] = sizes[0..2]
                @unk2 = 0
                for i in (0...10)
                        @names[i] = text_conv("UTF-16BE", "UTF-8", names[i]) # Multi-language names
                end
                @hash = "\0" * 16
                imet = self.pack()
                @hash = @crypt.createMD5Hash(imet[0,@size])
                @data = self.pack() + @data
                return @data
        end
