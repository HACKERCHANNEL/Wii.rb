        def encryptData(key, iv, data, doalign = true)
                if ((data.length() % @align) != 0) and doalign
                        data = data + ("\x00" * (align(data.length(), @align) - data.length()))
                end
                @crypter.encrypt
                @crypter.key = key
                @crypter.iv = iv
                e = @crypter.update(data)
                e << @crypter.final
                return e
        end
