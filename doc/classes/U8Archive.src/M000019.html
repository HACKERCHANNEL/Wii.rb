        def _dumpDir(dirname)
                files = @filearray.clone
                while true
                        file = files.shift
                        break if !file
                        data = files.shift
                        size = files.shift
                        if $DEBUG == true
                                puts file
                        end
                        if data == nil # Folder
                                if $DEBUG == true
                                        puts "Folder"
                                end
                                if not File.directory?(file)
                                        Dir.mkdir(file)
                                end
                        else
                                if $DEBUG == true
                                        puts "File"
                                end
                                outf = File.new(file, "w+")
                                outf.write(data[0,size])
                                outf.close
                        end
                end
        end
