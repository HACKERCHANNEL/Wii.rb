        def _loadDir(dirname)
                dirs = File.join("**", "*")
                dirlist = Dir.glob(dirs)
                while true
                        dir = dirlist.shift
                        break if !dir
                        if $DEBUG
                                puts dir
                        end
                        @filearray.push(dir)
                        if File.directory?(dir)
                                if $DEBUG
                                        puts "  Dir"
                                end
                                @files[dir] = nil
                                @filesizes[dir] = 0
                                @filearray.push(nil)
                                @filearray.push(0)
                        else
                                if $DEBUG
                                        puts "  File"
                                end
                                @files[dir] = readFile(dir)
                                @filesizes[dir] = File.size(dir)
                                @filearray.push(readFile(dir))
                                @filearray.push(File.size(dir))
                        end
                end
        end
