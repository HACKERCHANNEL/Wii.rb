        def compress(type)
                uncomp_len = @data.length()
                comp_type = type
                hdr  = uncomp_len << 8
                hdr |= comp_type << 4
                
                if comp_type == Compression::TYPE_LZ77
                        return lz77(hdr)
                else
                        raise ArgumentError, "Compression type not supported " + comp_type.to_s
                end
        end
