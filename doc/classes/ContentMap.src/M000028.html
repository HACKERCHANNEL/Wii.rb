        def load(data)
                @count = data.length() / 28
                for i in (0...@count)
                        entry = data[i*28,(i+1)*28].unpack("a8a20")
                        @namemap[entry[0]] = entry[1]
                        @hashmap[entry[1]] = entry[0]
                        array[i*2 + 0] = entry[0]

                end
